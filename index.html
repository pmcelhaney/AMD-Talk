IIFE

Immediately Invoked Function Expression

        
(function () {
    var settings = {  
        min: 100,
        max: 500,
        unit: 'lb'
    };
    
    ...
    
    window.Ruler = { ... };

})();

Immediately Invoked Function Expression

        
(function () {

    ...

})();

IIFE

Immediately Invoked Function Expression

        
(function ($, _, unitConverter) {

    ...
    
    window.Ruler = { ... };

})(window.jQuery, window._, window.UNITS);

Immediately Invoked Function Expression

(A)IIFE

(Almost) Immediately Invoked Function Expression


loadScripts_then_callFunction(
    [load('jquery.js'), load('underscore.js'), load('units.js')], 
    function ($, _, unitConverter) {

        ...
    
        window.Ruler = { ... };
    
    }
)

* foo.js must set window['foo']
